import os
import datetime

os.getcwd() # retorna o diretorio onde o programa esta rodando

os.listdir("path") # lista tudo que se encontra no diretorio passado

os.chdir("path/..") # muda o diretorio onde o programa esta rodando

os.mkdir("dir") # cria um diretorio

os.rmdir("dir") # remove diretorio

os.environ.get("HOME","") # Vai procurar a variavel de ambiente com o nome passado , o get permite retornar alguma coisa que não seja erro caso não encontre, no caso uma string vazia

os.path.isfile("arquivo.txt") # verifica se o arquivo existe
# True ou False

os.path.isdir() # verifica se o diretorio existe

os.path.abspath("arquivo.txt") # O diretorio onde o arquivo se encontra
#D:/users/arthu/arquivo.txt

os.pathsep # retorna uma string contendo o separador de path correspondente ao systema operacional usado

os.path.getsize("arquivo.txt") # tamanho em bytes
#0

os.rename("novel.txt","lixo.txt")

os.path.exists("lixo.txt")
#True ou False

os.remove("lixo.txt") # deletar o arquivo

timestamp = os.path.getmtime("arquivo.txt") # a ultima vez que foi modificado em timestamp

datetime.datetime.fromtimestamp(timestamp) # vai devolver uma data legivel
